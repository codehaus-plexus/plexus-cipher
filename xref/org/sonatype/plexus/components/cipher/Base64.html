33  /**
34   * <p>Provides Base64 encoding and decoding as defined by RFC 2045.</p>
35   * 
36   * <p>This class implements section <cite>6.8. Base64 Content-Transfer-Encoding</cite>
37   * from RFC 2045 <cite>Multipurpose Internet Mail Extensions (MIME) Part One:
38   * Format of Internet Message Bodies</cite> by Freed and Borenstein.</p>
39   *
40   * @author Apache Software Foundation
41   * @see <a href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>
42   * @since 1.0-dev
43   */
44  public class Base64
45  {
46  
47      /**
48       * Chunk size per RFC 2045 section 6.8.
49       * <p/>
50       * <p>The character limit does not count the trailing CRLF, but counts
51       * all other characters, including any equal signs.</p>
52       *
53       * @see <a href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045 section 6.8</a>
54       */
55      static final int CHUNK_SIZE = 76;
56  
57      /**
58       * Chunk separator per RFC 2045 section 2.1.
59       *
60       * @see <a href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045 section 2.1</a>
61       */
62      static final byte[] CHUNK_SEPARATOR = "\r\n".getBytes();
63  
64      /** The base length. */
65      static final int BASELENGTH = 255;
66  
67      /** Lookup length. */
68      static final int LOOKUPLENGTH = 64;
69  
70      /** Used to calculate the number of bits in a byte. */
71      static final int EIGHTBIT = 8;
72  
73      /** Used when encoding something which has fewer than 24 bits. */
74      static final int SIXTEENBIT = 16;
75  
76      /** Used to determine how many bits data contains. */
77      static final int TWENTYFOURBITGROUP = 24;
78  
79      /** Used to get the number of Quadruples. */
80      static final int FOURBYTE = 4;
81  
82      /** Used to test the sign of a byte. */
83      static final int SIGN = -128;
84  
85      /** Byte used to pad output. */
86      static final byte PAD = (byte) '=';
87  
88      /**
89       * Contains the Base64 values <code>0</code> through <code>63</code> accessed by using character encodings as
90       * indices.
91       * <p/>
92       * For example, <code>base64Alphabet['+']</code> returns <code>62</code>.
93       * </p>
94       * <p/>
95       * The value of undefined encodings is <code>-1</code>.
96       * </p>
97       */
98      private static final byte[] base64Alphabet = new byte[BASELENGTH];
99  
100     /**
101      * <p/>
102      * Contains the Base64 encodings <code>A</code> through <code>Z</code>, followed by <code>a</code> through
103      * <code>z</code>, followed by <code>0</code> through <code>9</code>, followed by <code>+</code>, and
104      * <code>/</code>.
105      * </p>
106      * <p/>
107      * This array is accessed by using character values as indices.
108      * </p>
109      * <p/>
110      * For example, <code>lookUpBase64Alphabet[62] </code> returns <code>'+'</code>.
111      * </p>
112      */
113     private static final byte[] lookUpBase64Alphabet = new byte[LOOKUPLENGTH];
114 
115     // Populating the lookup and character arrays
116     static {
117         for (int i = 0; i < BASELENGTH; i++) {
118             base64Alphabet[i] = (byte) -1;
119         }
120         for (int i = 'Z'; i >= 'A'; i--) {
121             base64Alphabet[i] = (byte) (i - 'A');
122         }
123         for (int i = 'z'; i >= 'a'; i--) {
124             base64Alphabet[i] = (byte) (i - 'a' + 26);
125         }
126         for (int i = '9'; i >= '0'; i--) {
127             base64Alphabet[i] = (byte) (i - '0' + 52);
128         }
129 
130         base64Alphabet['+'] = 62;
131         base64Alphabet['/'] = 63;
132 
133         for (int i = 0; i <= 25; i++) {
134             lookUpBase64Alphabet[i] = (byte) ('A' + i);
135         }
136 
137         for (int i = 26, j = 0; i <= 51; i++, j++) {
138             lookUpBase64Alphabet[i] = (byte) ('a' + j);
139         }
140 
141         for (int i = 52, j = 0; i <= 61; i++, j++) {
142             lookUpBase64Alphabet[i] = (byte) ('0' + j);
143         }
144 
145         lookUpBase64Alphabet[62] = (byte) '+';
146         lookUpBase64Alphabet[63] = (byte) '/';
147     }
148 
149     /**
150      * Returns whether or not the <code>octect</code> is in the base 64 alphabet.
151      *
152      * @param b The value to test
153      * @return <code>true</code> if the value is defined in the the base 64 alphabet, <code>false</code> otherwise.
154      */
155     public static boolean isBase64(byte b) {
156         return (b == PAD) || (b >= 0 && base64Alphabet[b] >= 0);
157     }
158 
159     /**
160      * Tests a given byte array to see if it contains
161      * only valid characters within the Base64 alphabet.
162      *
163      * @param arrayOctect byte array to test
164      * @return <code>true</code> if all bytes are valid characters in the Base64
165      *         alphabet or if the byte array is empty; false, otherwise
166      */
167     public static boolean isArrayByteBase64(byte[] arrayOctect) {
168 
169         arrayOctect = discardWhitespace(arrayOctect);
170 
171         int length = arrayOctect.length;
172         if (length == 0) {
173             // shouldn't a 0 length array be valid base64 data?
174             // return false;
175             return true;
176         }
177         for (int i = 0; i < length; i++) {
178             if (!isBase64(arrayOctect[i])) {
179                 return false;
180             }
181         }
182         return true;
183     }

208     /**
209      * Decodes an Object using the base64 algorithm.  This method
210      * is provided in order to satisfy the requirements of the
211      * Decoder interface, and will throw a DecoderException if the
212      * supplied object is not of type byte[].
213      *
214      * @param pObject Object to decode
215      * @return An object (of type byte[]) containing the
216      *         binary data which corresponds to the byte[] supplied.
217      * @throws IllegalArgumentException if the parameter supplied is not
218      *                                  of type byte[]
219      */
220     public Object decode(Object pObject) throws IllegalArgumentException {
221         if (!(pObject instanceof byte[])) {
222             throw new IllegalArgumentException("Parameter supplied to Base64 decode is not a byte[]");
223         }
224         return decode((byte[]) pObject);
225     }

228      * Decodes a byte[] containing containing
229      * characters in the Base64 alphabet.

232      * @return a byte array containing binary data

234     public byte[] decode(byte[] pArray) {
235         return decodeBase64(pArray);

238     /**
239      * Encodes binary data using the base64 algorithm, optionally
240      * chunking the output into 76 character blocks.
241      *
242      * @param binaryData Array containing binary data to encode.
243      * @param isChunked  if <code>true</code> this encoder will chunk
244      *                   the base64 output into 76 character blocks
245      * @return Base64-encoded data.
246      */
247     public static byte[] encodeBase64(byte[] binaryData, boolean isChunked) {
248         int lengthDataBits = binaryData.length * EIGHTBIT;
249         int fewerThan24bits = lengthDataBits % TWENTYFOURBITGROUP;
250         int numberTriplets = lengthDataBits / TWENTYFOURBITGROUP;
251         byte encodedData[];
252         int encodedDataLength;
253         int nbrChunks = 0;
254 
255         if (fewerThan24bits != 0) {
256             //data not divisible by 24 bit
257             encodedDataLength = (numberTriplets + 1) * 4;
258         } else {
259             // 16 or 8 bit
260             encodedDataLength = numberTriplets * 4;
261         }
262 
263         // If the output is to be "chunked" into 76 character sections,
264         // for compliance with RFC 2045 MIME, then it is important to
265         // allow for extra length to account for the separator(s)
266         if (isChunked) {
267 
268             nbrChunks =
269                 (CHUNK_SEPARATOR.length == 0 ? 0 : (int) Math.ceil((float) encodedDataLength / CHUNK_SIZE));
270             encodedDataLength += nbrChunks * CHUNK_SEPARATOR.length;
271         }
272 
273         encodedData = new byte[encodedDataLength];
274 
275         byte k, l, b1, b2, b3;
276         int dataIndex;
277         int i;
278         int encodedIndex = 0;
279         int nextSeparatorIndex = CHUNK_SIZE;
280         int chunksSoFar = 0;
281 
282         //log.debug("number of triplets = " + numberTriplets);
283         for (i = 0; i < numberTriplets; i++) {
284             dataIndex = i * 3;
285             b1 = binaryData[dataIndex];
286             b2 = binaryData[dataIndex + 1];
287             b3 = binaryData[dataIndex + 2];
288 
289             //log.debug("b1= " + b1 +", b2= " + b2 + ", b3= " + b3);
290 
291             l = (byte) (b2 & 0x0f);
292             k = (byte) (b1 & 0x03);
293 
294             byte val1 =
295                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
296             byte val2 =
297                 ((b2 & SIGN) == 0) ? (byte) (b2 >> 4) : (byte) ((b2) >> 4 ^ 0xf0);
298             byte val3 =
299                 ((b3 & SIGN) == 0) ? (byte) (b3 >> 6) : (byte) ((b3) >> 6 ^ 0xfc);
300 
301             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
302             //log.debug( "val2 = " + val2 );
303             //log.debug( "k4   = " + (k<<4) );
304             //log.debug(  "vak  = " + (val2 | (k<<4)) );
305             encodedData[encodedIndex + 1] =
306                 lookUpBase64Alphabet[val2 | (k << 4)];
307             encodedData[encodedIndex + 2] =
308                 lookUpBase64Alphabet[(l << 2) | val3];
309             encodedData[encodedIndex + 3] = lookUpBase64Alphabet[b3 & 0x3f];
310 
311             encodedIndex += 4;
312 
313             // If we are chunking, let's put a chunk separator down.
314             if (isChunked) {
315                 // this assumes that CHUNK_SIZE % 4 == 0
316                 if (encodedIndex == nextSeparatorIndex) {
317                     System.arraycopy(CHUNK_SEPARATOR,
318                         0,
319                         encodedData,
320                         encodedIndex,
321                         CHUNK_SEPARATOR.length);
322                     chunksSoFar++;
323                     nextSeparatorIndex =
324                         (CHUNK_SIZE * (chunksSoFar + 1)) +
325                         (chunksSoFar * CHUNK_SEPARATOR.length);
326                     encodedIndex += CHUNK_SEPARATOR.length;
327                 }
328             }
329         }
330 
331         // form integral number of 6-bit groups
332         dataIndex = i * 3;
333 
334         if (fewerThan24bits == EIGHTBIT) {
335             b1 = binaryData[dataIndex];
336             k = (byte) (b1 & 0x03);
337             //log.debug("b1=" + b1);
338             //log.debug("b1<<2 = " + (b1>>2) );
339             byte val1 =
340                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
341             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
342             encodedData[encodedIndex + 1] = lookUpBase64Alphabet[k << 4];
343             encodedData[encodedIndex + 2] = PAD;
344             encodedData[encodedIndex + 3] = PAD;
345         } else if (fewerThan24bits == SIXTEENBIT) {
346 
347             b1 = binaryData[dataIndex];
348             b2 = binaryData[dataIndex + 1];
349             l = (byte) (b2 & 0x0f);
350             k = (byte) (b1 & 0x03);
351 
352             byte val1 =
353                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
354             byte val2 =
355                 ((b2 & SIGN) == 0) ? (byte) (b2 >> 4) : (byte) ((b2) >> 4 ^ 0xf0);
356 
357             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
358             encodedData[encodedIndex + 1] =
359                 lookUpBase64Alphabet[val2 | (k << 4)];
360             encodedData[encodedIndex + 2] = lookUpBase64Alphabet[l << 2];
361             encodedData[encodedIndex + 3] = PAD;
362         }
363 
364         if (isChunked) {
365             // we also add a separator to the end of the final chunk.
366             if (chunksSoFar < nbrChunks) {
367                 System.arraycopy(CHUNK_SEPARATOR,
368                     0,
369                     encodedData,
370                     encodedDataLength - CHUNK_SEPARATOR.length,
371                     CHUNK_SEPARATOR.length);
372             }
373         }
374 
375         return encodedData;
376     }

378     /**
379      * Decodes Base64 data into octects
380      *
381      * @param base64Data Byte array containing Base64 data
382      * @return Array containing decoded data.
383      */
384     public static byte[] decodeBase64(byte[] base64Data) {
385         // RFC 2045 requires that we discard ALL non-Base64 characters
386         base64Data = discardNonBase64(base64Data);
387 
388         // handle the edge case, so we don't have to worry about it later
389         if (base64Data.length == 0) {
390             return new byte[0];
391         }
392 
393         int numberQuadruple = base64Data.length / FOURBYTE;
394         byte decodedData[];
395         byte b1, b2, b3, b4, marker0, marker1;
396 
397         // Throw away anything not in base64Data
398         int dataIndex;
399         int encodedIndex = 0;
400         {
401             // this sizes the output array properly - rlw
402             int lastData = base64Data.length;
403             // ignore the '=' padding
404             while (base64Data[lastData - 1] == PAD) {
405                 if (--lastData == 0) {
406                     return new byte[0];
407                 }
408             }
409             decodedData = new byte[lastData - numberQuadruple];
410         }
411 
412         for (int i = 0; i < numberQuadruple; i++) {
413             dataIndex = i * 4;
414             marker0 = base64Data[dataIndex + 2];
415             marker1 = base64Data[dataIndex + 3];
416 
417             b1 = base64Alphabet[base64Data[dataIndex]];
418             b2 = base64Alphabet[base64Data[dataIndex + 1]];
419 
420             if (marker0 != PAD && marker1 != PAD) {
421                 //No PAD e.g 3cQl
422                 b3 = base64Alphabet[marker0];
423                 b4 = base64Alphabet[marker1];
424 
425                 decodedData[encodedIndex] = (byte) (b1 << 2 | b2 >> 4);
426                 decodedData[encodedIndex + 1] =
427                     (byte) (((b2 & 0xf) << 4) | ((b3 >> 2) & 0xf));
428                 decodedData[encodedIndex + 2] = (byte) (b3 << 6 | b4);
429             } else if (marker0 == PAD) {
430                 //Two PAD e.g. 3c[Pad][Pad]
431                 decodedData[encodedIndex] = (byte) (b1 << 2 | b2 >> 4);
432             } else // if ( marker1 == PAD ) (always true at this point)
433             {
434                 //One PAD e.g. 3cQ[Pad]
435                 b3 = base64Alphabet[marker0];
436 
437                 decodedData[encodedIndex] = (byte) (b1 << 2 | b2 >> 4);
438                 decodedData[encodedIndex + 1] =
439                     (byte) (((b2 & 0xf) << 4) | ((b3 >> 2) & 0xf));
440             }
441             encodedIndex += 3;
442         }
443         return decodedData;
444     }

446     /**
447      * Discards any whitespace from a base-64 encoded block.
448      *
449      * @param data The base-64 encoded data to discard the whitespace
450      *             from.
451      * @return The data, less whitespace (see RFC 2045).
452      */
453     static byte[] discardWhitespace(byte[] data) {
454         byte groomedData[] = new byte[data.length];
455         int bytesCopied = 0;
456 
457         for (int i = 0; i < data.length; i++) {
458             switch (data[i]) {
459                 case (byte) ' ':
460                 case (byte) '\n':
461                 case (byte) '\r':
462                 case (byte) '\t':
463                     break;
464                 default:
465                     groomedData[bytesCopied++] = data[i];
466             }
467         }
468 
469         byte packedData[] = new byte[bytesCopied];
470 
471         System.arraycopy(groomedData, 0, packedData, 0, bytesCopied);
472 
473         return packedData;
474     }

476     /**
477      * Discards any characters outside of the base64 alphabet, per
478      * the requirements on page 25 of RFC 2045 - "Any characters
479      * outside of the base64 alphabet are to be ignored in base64
480      * encoded data."
481      *
482      * @param data The base-64 encoded data to groom
483      * @return The data, less non-base64 characters (see RFC 2045).
484      */
485     static byte[] discardNonBase64(byte[] data) {
486         byte groomedData[] = new byte[data.length];
487         int bytesCopied = 0;
488 
489         for (int i = 0; i < data.length; i++) {
490             if (isBase64(data[i])) {
491                 groomedData[bytesCopied++] = data[i];
492             }
493         }
494 
495         byte packedData[] = new byte[bytesCopied];
496 
497         System.arraycopy(groomedData, 0, packedData, 0, bytesCopied);
498 
499         return packedData;
500     }
